\subsection{Centered Figure}

\begin{figure}[h]
  \centering
  \includegraphics[width=0.5\textwidth]{image-1}
  \caption{This is a centered figure}
  \label{fig:image1}
\end{figure}

\begin{lstlisting}
  \begin{figure}[h]
    \centering
    \includegraphics[width=0.5\textwidth]{<file-name>}
    \caption{This is a centered figure}
    \label{fig:<yourFigureLabel>}
  \end{figure}
\end{lstlisting}
